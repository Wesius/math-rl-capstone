% ---- SLIDE 3: Method ----
\begin{frame}{Method: GRPO + LoRA}
    \textbf{Group Relative Policy Optimization} (DeepSeekMath, 2024)

    \begin{enumerate}
        \item Generate $G = 8$ completions per prompt
        \item Score each with reward function: $\text{eval}(e) = n$ ?
        \item Compute group-relative advantage: $\hat{A}_i = \frac{r_i - \text{mean}(\mathbf{r})}{\text{std}(\mathbf{r})}$
        \item Update policy with clipped surrogate loss
    \end{enumerate}

    \vspace{1em}
    \textbf{LoRA} fine-tuning: rank 32, targeting attention + MLP layers

    \vspace{0.5em}
    $\rightarrow$ Trains $\sim$2--3\% of parameters. No reward model needed.
\end{frame}

% ---- SLIDE 4: Reward ----
\begin{frame}{Reward Design}
    \begin{table}
        \centering
        \begin{tabular}{llcl}
            \toprule
            \textbf{Component} & \textbf{What} & \textbf{Weight} & \textbf{Why} \\
            \midrule
            Correctness & eval(expr) == target & 2.0 & Primary objective \\
            Closeness & $1 - |r - n|/|n|$ & 0.5 & Gradient shaping \\
            Format & \texttt{<expr>} tags & 0.3 & Structured output \\
            Complexity & Multi-operator bonus & 0.2 & Avoids trivial \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{1em}
    \textbf{Key insight:} Closeness reward prevents sparse gradients early in training when the model rarely hits exact targets.
\end{frame}

% ---- SLIDE 10: Takeaways ----
\begin{frame}{Key Takeaways}
    \begin{enumerate}
        \item \textbf{Reward hacking is multi-layered, but solvable} --- identity $\to$ formula $\to$ syntax-only $\to$ genuine reasoning
        \item \textbf{Structural constraints beat incentives} --- banning the target number forced real decomposition
        \item \textbf{Diversity is measurable} --- template ratio, edit distance, operator entropy quantify it
        \item \textbf{350M params is enough} --- discovered 2\^{}8=256 on its own with the right reward signal
    \end{enumerate}

    \vspace{1em}
    \textbf{Future work:} Longer training, curriculum learning, integer-only bonus, architecture comparison
\end{frame}
